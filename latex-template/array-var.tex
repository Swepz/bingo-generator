  {% #1 = name, #2 = items
    \clist_clear_new:c { array_#1_clist }
    \clist_set:cn { array_#1_clist } { #2 }
  }

  {% #1 = name, #2 = row index, #3 = column index
    \clist_item:en
    {
      \clist_item:cn { array_#1_clist } { #2 }
    }
    { #3 }
  }
  \cs_generate_variant:Nn \clist_item:nn { e }
